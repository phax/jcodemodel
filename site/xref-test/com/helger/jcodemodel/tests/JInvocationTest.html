41  package com.helger.jcodemodel.tests;
42  
43  import java.io.IOException;
44  import java.math.BigInteger;
45  import java.util.ArrayList;
46  import java.util.Comparator;
47  import java.util.List;
48  
49  import org.junit.Test;
50  
51  import com.helger.jcodemodel.JClassAlreadyExistsException;
52  import com.helger.jcodemodel.JCodeModel;
53  import com.helger.jcodemodel.JDefinedClass;
54  import com.helger.jcodemodel.JExpr;
55  import com.helger.jcodemodel.JMethod;
56  import com.helger.jcodemodel.JMod;
57  import com.helger.jcodemodel.JTypeVar;
58  import com.helger.jcodemodel.writer.SingleStreamCodeWriter;
59  
60  public class JInvocationTest
61  {
62    @Test
63    public void testWithGenerics () throws JClassAlreadyExistsException, IOException
64    {
65      final JCodeModel cm = new JCodeModel ();
66      final JDefinedClass cls = cm._class ("TestInvocation");
67      final JTypeVar tc = cls.generify ("IMPL");
68  
69      final JMethod mc = cls.constructor (JMod.PUBLIC);
70      mc.param (JMod.FINAL, tc, "ctor");
71  
72      final JMethod m1 = cls.method (JMod.PUBLIC, cm.VOID, "foo1");
73      final JTypeVar tv1 = m1.generify ("T");
74      m1.param (JMod.FINAL, tv1, "foo");
75      m1.body ()._return ();
76  
77      final JMethod m1a = cls.method (JMod.PUBLIC, cm.VOID, "foo1a");
78      final JTypeVar tv1a = m1a.generify ("T", BigInteger.class);
79      m1a.param (JMod.FINAL, tv1a, "foo");
80      m1a.body ()._return ();
81  
82      final JMethod m1b = cls.method (JMod.PUBLIC, cm.VOID, "foo1b");
83      m1b.param (JMod.FINAL, cm.ref (Comparator.class).narrow (cm.ref (CharSequence.class).wildcardSuper ()), "foo");
84      m1b.body ()._return ();
85  
86      final JMethod m2 = cls.method (JMod.PUBLIC, cm.VOID, "foo2");
87      final JTypeVar tv21 = m2.generify ("T");
88      final JTypeVar tv22 = m2.generify ("U");
89      final JTypeVar tv23 = m2.generify ("V");
90      m2.param (JMod.FINAL, tv21, "t");
91      m2.param (JMod.FINAL, tv22, "u");
92      m2.param (JMod.FINAL, tv23, "v");
93      m2.body ()._return ();
94  
95      final JMethod minvoke = cls.method (JMod.PUBLIC, cm.VOID, "bar");
96      minvoke.body ()._new (cls).narrow (Integer.class).arg (cm.INT.wrap (JExpr.lit (17)));
97      minvoke.body ().invokeThis (m1).narrow (String.class).arg ("jippie");
98      minvoke.body ().invoke (m1).arg ("jippie");
99      minvoke.body ()
100            .invokeThis (m2)
101            .narrow (String.class)
102            .narrow (cls)
103            .narrow (cm.ref (List.class).narrow (Long.class))
104            .arg ("jippie")
105            .arg (JExpr._this ())
106            .arg (JExpr._new (cm.ref (ArrayList.class).narrow (Long.class)));
107     minvoke.body ()
108            .invoke (m2)
109            .arg ("jippie")
110            .arg (JExpr._this ())
111            .arg (JExpr._new (cm.ref (ArrayList.class).narrow (Long.class)));
112 
113     cm.build (new SingleStreamCodeWriter (System.out));
114   }
115 }
