204   /**
205    * Get the content of the code model as a byte array in
206    * {@link #DEFAULT_ENCODING}
207    *
208    * @param cm
209    *        Source code model
210    * @return The byte array
211    */
212   @Nonnull
213   public static byte [] getAllBytes (@Nonnull final JCodeModel cm)
214   {
215     try (final ByteArrayOutputStream aBAOS = new ByteArrayOutputStream ())
216     {
217       new JCMWriter (cm).build (new OutputStreamCodeWriter (aBAOS, DEFAULT_ENCODING));
218       return aBAOS.toByteArray ();
219     }
220     catch (final IOException ex)
221     {
222       throw new UncheckedIOException (ex);
223     }
224   }

226   @Nonnull
227   private static CompilationUnit _parseWithJavaParser (final byte [] aBytes,
228                                                        final JPackage aPackage,
229                                                        final String sFilename) throws IOException
230   {
231     if (false)
232       System.out.println (new String (aBytes, DEFAULT_ENCODING));
233 
234     System.out.println ("Parsing " +
235                         (aPackage == null || aPackage.isUnnamed () ? "" : aPackage.name () + ".") +
236                         (sFilename.endsWith (".java") ? sFilename.substring (0, sFilename.length () - 5) : sFilename));
237 
238     try (final ByteArrayInputStream bis = new ByteArrayInputStream (aBytes))
239     {
240       // Parse what was written
241       final ParseResult <CompilationUnit> ret = new JavaParser ().parse (bis, DEFAULT_ENCODING);
242       return ret.getResult ().get ();
243     }
244   }

246   @Nonnull
247   private static org.eclipse.jdt.core.dom.CompilationUnit _parseWithJDT (final String sUnitName, final char [] aCode)
248   {
249     final ASTParser parser = ASTParser.newParser (AST.JLS11);
250     parser.setResolveBindings (true);
251     parser.setStatementsRecovery (true);
252     parser.setBindingsRecovery (true);
253     parser.setKind (ASTParser.K_COMPILATION_UNIT);
254     parser.setSource (aCode);
255     parser.setUnitName (sUnitName);
256     final Map <String, String> aOptions = new HashMap <> ();
257     aOptions.put (CompilerOptions.OPTION_Source, "1.8");
258     parser.setCompilerOptions (aOptions);
259 
260     final IProgressMonitor aPM = null;
261     final org.eclipse.jdt.core.dom.CompilationUnit astRoot = (org.eclipse.jdt.core.dom.CompilationUnit) parser.createAST (aPM);
262     if (astRoot == null)
263       throw new IllegalStateException ("Failed to compile:\n" + new String (aCode));
264     if (false)
265       System.out.println (astRoot.toString ());
266     final IProblem [] aProblems = astRoot.getProblems ();
267     if (aProblems != null && aProblems.length > 0)
268       throw new IllegalStateException ("Compilation problems " + Arrays.toString (aProblems));
269     return astRoot;
270   }

272   public static void parseCodeModel (@Nonnull final JCodeModel cm)
273   {
274     try
275     {
276       new JCMWriter (cm).build (new AbstractCodeWriter (DEFAULT_ENCODING, "\n")
277       {
278         @Override
279         public OutputStream openBinary (final JPackage aPackage, final String sFilename) throws IOException
280         {
281           return new ByteArrayOutputStream ()
282           {
283             @Override
284             public void close () throws IOException
285             {
286               super.close ();
287 
288               final byte [] aBytes = toByteArray ();
289 
290               // Get result as bytes and parse
291               _parseWithJavaParser (aBytes, aPackage, sFilename);
292               _parseWithJDT (sFilename, new String (aBytes, DEFAULT_ENCODING).toCharArray ());
293             }
294           };
295         }
296 
297         @Override
298         public void close () throws IOException
299         {}
300       });
301     }
302     catch (final IOException ex)
303     {
304       throw new UncheckedIOException (ex);
305     }
306   }

308   @Nonnegative
309   private static int _count (final Iterator <?> x)
310   {
311     int ret = 0;
312     while (x.hasNext ())
313     {
314       x.next ();
315       ret++;
316     }
317     return ret;
318   }

320   @Nonnull
321   public static CompilationUnit parseAndGetSingleClassCodeModel (@Nonnull final JCodeModel cm)
322   {
323     assert cm != null;
324     assert _count (cm.packages ()) == 1;
325     assert cm.packages ().next ().classes ().size () == 1;
326     final byte [] aBytes = getAllBytes (cm);
327     try
328     {
329       return _parseWithJavaParser (aBytes, null, "full-jcodemodel");
330     }
331     catch (final IOException ex)
332     {
333       throw new UncheckedIOException (ex);
334     }
335   }
