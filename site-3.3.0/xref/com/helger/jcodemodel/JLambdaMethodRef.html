41  package com.helger.jcodemodel;

43  import javax.annotation.Nonnull;
44  import javax.annotation.Nullable;

46  import com.helger.jcodemodel.util.JCValueEnforcer;

48  /**
49   * This is a single Java 8 lambda method reference expression.
50   *
51   * @author Philip Helger
52   * @since 2.8.3
53   */
54  public class JLambdaMethodRef implements IJExpression
55  {
56    private final boolean m_bStatic;
57    private final AbstractJType m_aType;
58    private final JVar m_aVar;
59    private final IJExpression m_aLhsExpr;
60    private final JMethod m_aMethod;
61    private final String m_sMethodName;
62  
63    /**
64     * Constructor to reference the passed static method. It uses the name of the
65     * owning class as base (<code>owning::method</code>).
66     *
67     * @param aMethod
68     *        The static method to reference. May not be <code>null</code>.
69     * @throws IllegalArgumentException
70     *         If the passed method is not static
71     */
72    public JLambdaMethodRef (@Nonnull final JMethod aMethod)
73    {
74      JCValueEnforcer.notNull (aMethod, "Method");
75      JCValueEnforcer.isTrue (aMethod.mods ().isStatic (),
76                              "Only static methods can be used with this constructor. Use the constructor with JVar for instance methods.");
77  
78      m_bStatic = true;
79      m_aType = aMethod.owningClass ();
80      m_aVar = null;
81      m_aLhsExpr = null;
82      m_aMethod = aMethod;
83      m_sMethodName = null;
84    }
85  
86    /**
87     * Constructor for a static constructor method reference
88     * (<code>type::new</code>).
89     *
90     * @param aType
91     *        Type to reference the constructor from. May not be <code>null</code>
92     *        .
93     * @deprecated Use the factory method {@link #createForNew(AbstractJType)}
94     *             instead.
95     */
96    @Deprecated
97    public JLambdaMethodRef (@Nonnull final AbstractJType aType)
98    {
99      this (aType, "new");
100   }
101 
102   /**
103    * Constructor for an arbitrary static method reference
104    * (<code>type::name</code>).
105    *
106    * @param aType
107    *        Type the method belongs to. May not be <code>null</code>.
108    * @param sMethod
109    *        Name of the static method to reference. May neither be
110    *        <code>null</code> nor empty.
111    */
112   public JLambdaMethodRef (@Nonnull final AbstractJType aType, @Nonnull final String sMethod)
113   {
114     JCValueEnforcer.notNull (aType, "Type");
115     JCValueEnforcer.notEmpty (sMethod, "Method");
116 
117     m_bStatic = true;
118     m_aType = aType;
119     m_aVar = null;
120     m_aLhsExpr = null;
121     m_aMethod = null;
122     m_sMethodName = sMethod;
123   }
124 
125   /**
126    * Constructor for an arbitrary instance method reference
127    * (<code>var::name</code>).
128    *
129    * @param aVar
130    *        Variable containing the instance. May not be <code>null</code>.
131    * @param sMethod
132    *        Name of the method to reference. May neither be <code>null</code>
133    *        nor empty.
134    */
135   public JLambdaMethodRef (@Nonnull final JVar aVar, @Nonnull final String sMethod)
136   {
137     JCValueEnforcer.notNull (aVar, "Var");
138     JCValueEnforcer.notEmpty (sMethod, "Method");
139 
140     m_bStatic = false;
141     m_aType = aVar.type ();
142     m_aVar = aVar;
143     m_aLhsExpr = null;
144     m_aMethod = null;
145     m_sMethodName = sMethod;
146   }
147 
148   /**
149    * Constructor for an arbitrary instance method reference
150    * (<code>var::name</code>).
151    *
152    * @param aVar
153    *        Variable containing the instance. May not be <code>null</code>.
154    * @param aMethod
155    *        The instance method to reference. May not be <code>null</code>.
156    */
157   public JLambdaMethodRef (@Nonnull final JVar aVar, @Nonnull final JMethod aMethod)
158   {
159     JCValueEnforcer.notNull (aVar, "Var");
160     JCValueEnforcer.notNull (aMethod, "Method");
161     JCValueEnforcer.isFalse (aMethod.mods ().isStatic (),
162                              "Only instance methods can be used with this constructor. Use the constructor with JMethod only for static methods.");
163 
164     m_bStatic = false;
165     m_aType = aVar.type ();
166     m_aVar = aVar;
167     m_aLhsExpr = null;
168     m_aMethod = aMethod;
169     m_sMethodName = null;
170   }
171 
172   /**
173    * Constructor for an arbitrary invocation method reference.<br>
174    * Note: in v3.3.0 the parameter type changed from {@link JInvocation} to
175    * {@link IJExpression}.
176    *
177    * @param aLhsExpr
178    *        Left hand side of the lambda expression. May not be
179    *        <code>null</code>.
180    * @param sMethod
181    *        Name of the method to reference. May neither be <code>null</code>
182    *        nor empty.
183    */
184   public JLambdaMethodRef (@Nonnull final IJExpression aLhsExpr, @Nonnull final String sMethod)
185   {
186     JCValueEnforcer.notNull (aLhsExpr, "Invocation");
187     JCValueEnforcer.notEmpty (sMethod, "Method");
188 
189     m_bStatic = false;
190     m_aType = null;
191     m_aVar = null;
192     m_aLhsExpr = aLhsExpr;
193     m_aMethod = null;
194     m_sMethodName = sMethod;
195   }
196 
197   /**
198    * Constructor for an arbitrary invocation method reference.<br>
199    * Note: in v3.3.0 the parameter type changed from {@link JInvocation} to
200    * {@link IJExpression}.
201    *
202    * @param aLhsExpr
203    *        Left hand side of the lambda expression. May not be
204    *        <code>null</code>.
205    * @param aMethod
206    *        The instance method to reference. May not be <code>null</code>.
207    */
208   public JLambdaMethodRef (@Nonnull final IJExpression aLhsExpr, @Nonnull final JMethod aMethod)
209   {
210     JCValueEnforcer.notNull (aLhsExpr, "Invocation");
211     JCValueEnforcer.notNull (aMethod, "Method");
212     JCValueEnforcer.isFalse (aMethod.mods ().isStatic (),
213                              "Only instance methods can be used with this constructor. Use the constructor with JMethod only for static methods.");
214 
215     m_bStatic = false;
216     m_aType = aMethod.owningClass ();
217     m_aVar = null;
218     m_aLhsExpr = aLhsExpr;
219     m_aMethod = aMethod;
220     m_sMethodName = null;
221   }
222 
223   /**
224    * @return <code>true</code> if this is a static reference, <code>false</code>
225    *         if this is an instance reference.
226    */
227   public boolean isStaticRef ()
228   {
229     return m_bStatic;
230   }
231 
232   /**
233    * @return The type owning the method. May be <code>null</code> if invoked
234    *         with another JInvocation.
235    */
236   @Nullable
237   public AbstractJType type ()
238   {
239     return m_aType;
240   }
241 
242   /**
243    * @return The variable for the instance reference. May be <code>null</code>
244    *         if this is a static or invocation reference.
245    */
246   @Nullable
247   public JVar var ()
248   {
249     return m_aVar;
250   }
251 
252   /**
253    * Note: v3.3.0 changed the return type to be {@link IJExpression}
254    *
255    * @return The invocation reference. May be <code>null</code> if this is a
256    *         static or variable reference.
257    * @deprecated Use {@link #lhsExpr()} instead
258    */
259   @Deprecated
260   @Nullable
261   public JInvocation invocation ()
262   {
263     return m_aLhsExpr instanceof JInvocation="../../../com/helger/jcodemodel/JInvocation.html#JInvocation">JInvocation ? (JInvocation) m_aLhsExpr : null;
264   }
265 
266   /**
267    * @return The left hand side expression. May be <code>null</code> if this is
268    *         a static or variable reference.
269    * @since 3.3.0
270    */
271   @Nullable
272   public IJExpression lhsExpr ()
273   {
274     return m_aLhsExpr;
275   }
276 
277   /**
278    * @return The owning method. May be <code>null</code> if a constructor with
279    *         method name was used.
280    */
281   @Nullable
282   public JMethod method ()
283   {
284     return m_aMethod;
285   }
286 
287   /**
288    * @return The name of the referenced method. Never <code>null</code>.
289    */
290   @Nonnull
291   public String methodName ()
292   {
293     return m_aMethod != null ? m_aMethod.name () : m_sMethodName;
294   }
295 
296   @Override
297   public void generate (@Nonnull final IJFormatter f)
298   {
299     if (isStaticRef ())
300       f.type (type ());
301     else
302       if (m_aVar != null)
303         f.generable (m_aVar);
304       else
305         f.generable (m_aLhsExpr);
306     f.print ("::").print (methodName ());
307   }
308 
309   /**
310    * Factory method for a static constructor method reference
311    * (<code>type::new</code>).
312    *
313    * @param aType
314    *        Type to reference the constructor from. May not be <code>null</code>
315    *        .
316    * @return The created object. Never <code>null</code>.
317    * @since 3.2.4
318    */
319   @Nonnull
320   public static JLambdaMethodRef createForNew (@Nonnull final AbstractJType aType)
321   {
322     return new JLambdaMethodRef (aType, "new");
323   }
