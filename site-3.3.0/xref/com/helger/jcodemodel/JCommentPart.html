41  package com.helger.jcodemodel;
42  
43  import java.util.ArrayList;
44  import java.util.Collection;
45  import java.util.Iterator;
46  
47  import javax.annotation.Nonnull;
48  import javax.annotation.Nullable;

50  /**
51   * A part is a part of a javadoc comment, and it is a list of values.
52   * <p>
53   * A part can contain a free-form text. This text is modeled as a collection of
54   * 'values' in this class. A value can be a {@link AbstractJType} (which will be
55   * prinited with a @link tag), anything that can be turned into a {@link String}
56   * via the {@link Object#toString()} method, or a {@link Collection}/array of
57   * those objects.
58   * <p>
59   * Values can be added through the various append methods one by one or in a
60   * bulk.
61   *
62   * @author Kohsuke Kawaguchi
63   */
64  public class JCommentPart extends ArrayList <Object>
65  {
66    public JCommentPart ()
67    {}
68  
69    /**
70     * Appends a new value.
71     *
72     * @param aValue
73     *        If the value is {@link AbstractJType} it will be printed as a @link
74     *        tag. Otherwise it will be converted to String via
75     *        {@link Object#toString()}.
76     * @return this for chaining
77     */
78    @Nonnull
79    public JCommentPart append (@Nullable final Object aValue)
80    {
81      add (aValue);
82      return this;
83    }
84  
85    @Override
86    public boolean add (@Nullable final Object aValue)
87    {
88      return _flattenAppend (aValue);
89    }
90  
91    @Nullable
92    public static String getHTMLEscaped (@Nullable final String sValue)
93    {
94      if (sValue == null)
95        return null;
96      if (sValue.length () == 0)
97        return "";
98  
99      final StringBuilder aSB = new StringBuilder (sValue.length () * 2);
100     for (final char c : sValue.toCharArray ())
101       if (c == '&')
102         aSB.append ("&amp;");
103       else
104         if (c == '<')
105           aSB.append ("&lt;");
106         else
107           if (c == '>')
108             aSB.append ("&gt;");
109           else
110             if (c == '"')
111               aSB.append ("&quot;");
112             else
113               aSB.append (c);
114     return aSB.toString ();
115   }
116 
117   /**
118    * Add the provided value, but before correct HTML character masking ("&lt;"
119    * becomes "&amp;lt;" etc.).
120    *
121    * @param sValue
122    *        Source value. May be <code>null</code>.
123    * @return <code>true</code> if added, <code>false</code> if the source is
124    *         <code>null</code>.
125    */
126   public boolean addMasked (@Nullable final String sValue)
127   {
128     final String sMasked = getHTMLEscaped (sValue);
129     if (sMasked == null)
130       return false;
131     return super.add (sMasked);
132   }
133 
134   private boolean _flattenAppend (@Nullable final Object aValue)
135   {
136     if (aValue == null)
137       return false;
138 
139     boolean bAny = false;
140     if (aValue instanceof Object [])
141     {
142       for (final Object o : (Object []) aValue)
143         if (_flattenAppend (o))
144           bAny = true;
145     }
146     else
147       if (aValue instanceof Collection <?>)
148       {
149         for (final Object o : (Collection <?>) aValue)
150           if (_flattenAppend (o))
151             bAny = true;
152       }
153       else
154       {
155         // Only String and AbstractJType are allowed
156         if (aValue instanceof String || aValue instanceof AbstractJType)
157           bAny = super.add (aValue);
158         else
159           throw new IllegalArgumentException ("Value is of an unsupported type: " + aValue.getClass ().toString ());
160       }
161     return bAny;
162   }

164   /**
165    * Writes this part into the formatter by using the specified indentation.
166    *
167    * @param f
168    *        Formatter to use
169    * @param sIndent
170    *        Indentation to use
171    */
172   protected void format (@Nonnull final IJFormatter f, final String sIndent)
173   {
174     if (!f.isPrinting ())
175     {
176       // quickly pass the types to IJFormatter, as that's all we care.
177       // we don't need to worry about the exact formatting of text.
178       for (final Object o : this)
179         if (o instanceof AbstractJClass)
180           f.generable ((AbstractJClass) o);
181       return;
182     }
183 
184     if (!isEmpty ())
185       f.print (sIndent);
186 
187     final Iterator <Object> itr = iterator ();
188     while (itr.hasNext ())
189     {
190       final Object o = itr.next ();
191 
192       if (o instanceof String)
193       {
194         int idx;
195         String sStr = (String) o;
196         while ((idx = sStr.indexOf ('\n')) != -1)
197         {
198           final String line = sStr.substring (0, idx);
199           if (line.length () > 0)
200             f.print (_escape (line));
201           sStr = sStr.substring (idx + 1);
202           f.newline ().print (sIndent);
203         }
204         if (sStr.length () != 0)
205           f.print (_escape (sStr));
206       }
207       else
208         if (o instanceof AbstractJClass)
209         {
210           // TODO: this doesn't print the parameterized type properly
211           ((AbstractJClass) o).printLink (f);
212         }
213         else
214           if (o instanceof AbstractJType)
215           {
216             f.generable ((AbstractJType) o);
217           }
218           else
219             throw new IllegalStateException ("Invalid type present: " + o);
220     }
221 
222     if (!isEmpty ())
223       f.newline ();
224   }

226   /**
227    * Escapes the appearance of the comment terminator "*" followed by "/". Note:
228    * quotes etc are not masked!
229    *
230    * @param sText
231    *        Source text. May not be <code>null</code>.
232    * @return Escaped text
233    */
234   @Nonnull
235   private static String _escape (@Nonnull final String sText)
236   {
237     String s = sText;
238     while (true)
239     {
240       final int nIdx = s.indexOf ("*/");
241       if (nIdx < 0)
242         break;
243       // Put an empty comment between "*" and "/"
244       s = s.substring (0, nIdx + 1) + "<!---->" + s.substring (nIdx + 1);
245     }
246     return s;
247   }
