41  package com.helger.jcodemodel;
42  
43  import static org.junit.Assert.assertEquals;
44  
45  import org.junit.Test;
46  
47  import com.helger.jcodemodel.util.CodeModelTestsHelper;
48  
49  public final class InnerClassFuncTest
50  {
51    @Test
52    public void innerClassesAreImported () throws Exception
53    {
54      final JCodeModel cm = new JCodeModel ();
55      final JDefinedClass aClass = cm._class ("org.test.DaTestClass");
56      final JDefinedClass daInner1 = aClass._class ("Inner");
57      final JDefinedClass daInnerInner = daInner1._class ("InnerInner");
58      final JDefinedClass daInner2 = aClass._class ("DaTestClassInner");
59      final JDefinedClass daInner2Inner = daInner2._class ("Inner2");
60  
61      assertEquals ("Inner", daInner1.name ());
62      assertEquals ("org.test.DaTestClass.Inner", daInner1.fullName ());
63      assertEquals ("org.test.DaTestClass$Inner", daInner1.binaryName ());
64  
65      assertEquals ("InnerInner", daInnerInner.name ());
66      assertEquals ("org.test.DaTestClass.Inner.InnerInner", daInnerInner.fullName ());
67      assertEquals ("org.test.DaTestClass$Inner$InnerInner", daInnerInner.binaryName ());
68  
69      aClass.method (JMod.PUBLIC, daInner1, "getInner");
70      aClass.method (JMod.PUBLIC, daInnerInner, "getInnerInner");
71      aClass.method (JMod.PUBLIC, daInner2, "getInner2");
72      aClass.method (JMod.PUBLIC, daInner2Inner, "getInner2Inner");
73  
74      final JDefinedClass otherClass = cm._class ("org.test.OtherClass");
75      otherClass.method (JMod.PUBLIC, daInner1, "getInner");
76      otherClass.method (JMod.PUBLIC, daInnerInner, "getInnerInner");
77      otherClass.method (JMod.PUBLIC, daInner2Inner, "getInner2Inner");
78      otherClass.method (JMod.PUBLIC, aClass, "getOuter");
79  
80      CodeModelTestsHelper.printCodeModel (cm);
81    }
82  }
