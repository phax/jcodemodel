295                                              @Nonnull final List <? extends AbstractJClass> aBindings)
296   {
297     final AbstractJClass b = m_aBasis.substituteParams (aVariables, aBindings);
298     boolean bDifferent = b != m_aBasis;
299 
300     final List <AbstractJClass> clazz = new ArrayList <> (m_aArgs.size ());
301     for (final AbstractJClass aClass : m_aArgs)
302     {
303       final AbstractJClass c = aClass.substituteParams (aVariables, aBindings);
304       clazz.add (c);
305       bDifferent |= c != aClass;
306     }
307 
308     if (bDifferent)
309       return new JNarrowedClass (b, clazz);
310     return this;
311   }
