41  package com.helger.jcodemodel;
42  
43  import java.io.Closeable;
44  import java.io.IOException;
45  import java.io.OutputStream;
46  import java.io.OutputStreamWriter;
47  import java.io.Writer;
48  import java.nio.charset.Charset;
49  import java.nio.charset.CharsetEncoder;
50  import java.util.BitSet;
51  
52  import javax.annotation.Nonnull;
53  import javax.annotation.Nullable;
54  
55  import com.helger.jcodemodel.util.UnicodeEscapeWriter;
56  
57  /**
58   * Receives generated code and writes to the appropriate storage.
59   *
60   * @author Kohsuke Kawaguchi (kohsuke.kawaguchi@sun.com)
61   */
62  public abstract class AbstractCodeWriter implements Closeable
63  {
64    private static final class JavaUnicodeEscapeWriter extends UnicodeEscapeWriter
65    {
66      private static final BitSet ESCAPE = new BitSet (128);
67  
68      static
69      {
70        for (int i = 0; i < 0x20; i++)
71          if (i != '\t' && i != '\r' && i != '\n')
72            ESCAPE.set (i, true);
73      }
74  
75      // can't change this signature to Encoder because
76      // we can't have Encoder in method signature
77      private final CharsetEncoder _encoder;
78  
79      private JavaUnicodeEscapeWriter (@Nonnull final OutputStreamWriter bw)
80      {
81        super (bw);
82        _encoder = Charset.forName (bw.getEncoding ()).newEncoder ();
83      }
84  
85      @Override
86      protected boolean requireEscaping (final int ch)
87      {
88        // control characters
89        if (ESCAPE.get (ch))
90          return true;
91  
92        // check ASCII chars, for better performance
93        if (ch < 0x80)
94          return false;
95  
96        return !_encoder.canEncode ((char) ch);
97      }
98    }
99  
100   /**
101    * Encoding to be used by the writer. Null means platform specific encoding.
102    */
103   private final Charset m_aEncoding;
104 
105   protected AbstractCodeWriter (@Nullable final Charset aEncoding)
106   {
107     m_aEncoding = aEncoding;
108   }
109 
110   @Nullable
111   public Charset encoding ()
112   {
113     return m_aEncoding;
114   }
115 
116   /**
117    * Called by CodeModel to store the specified file. The callee must allocate a
118    * storage to store the specified file. <br>
119    * The returned stream will be closed before the next file is stored. So the
120    * callee can assume that only one OutputStream is active at any given time.
121    *
122    * @param pkg
123    *        The package of the file to be written.
124    * @param fileName
125    *        File name without the path. Something like "Foo.java" or
126    *        "Bar.properties"
127    * @return OutputStream to write to
128    * @throws IOException
129    *         On IO error
130    */
131   @Nonnull
132   public abstract OutputStream openBinary (@Nonnull JPackage pkg, @Nonnull String fileName) throws IOException;
133 
134   /**
135    * Called by CodeModel to store the specified file. The callee must allocate a
136    * storage to store the specified file. <br>
137    * The returned stream will be closed before the next file is stored. So the
138    * callee can assume that only one OutputStream is active at any given time.
139    *
140    * @param pkg
141    *        The package of the file to be written.
142    * @param fileName
143    *        File name without the path. Something like "Foo.java" or
144    *        "Bar.properties"
145    * @return Writer to write to
146    * @throws IOException
147    *         On IO error
148    */
149   @Nonnull
150   public Writer openSource (@Nonnull final JPackage pkg, @Nonnull final String fileName) throws IOException
151   {
152     final OutputStream os = openBinary (pkg, fileName);
153     final OutputStreamWriter bw = m_aEncoding != null ? new OutputStreamWriter (os, m_aEncoding)
154                                                       : new OutputStreamWriter (os);
155 
156     // create writer
157     try
158     {
159       return new JavaUnicodeEscapeWriter (bw);
160     }
161     catch (final Throwable t)
162     {
163       return new UnicodeEscapeWriter (bw);
164     }
165   }
166 
167   /**
168    * Called by CodeModel at the end of the process.
169    */
170   public abstract void close () throws IOException;
171 }
