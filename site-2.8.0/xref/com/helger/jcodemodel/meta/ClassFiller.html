30  package com.helger.jcodemodel.meta;
31  
32  import java.util.List;
33  
34  import javax.lang.model.element.Element;
35  import javax.lang.model.element.ElementKind;
36  import javax.lang.model.element.ExecutableElement;
37  import javax.lang.model.element.TypeElement;
38  import javax.lang.model.element.TypeParameterElement;
39  import javax.lang.model.element.VariableElement;
40  import javax.lang.model.type.TypeKind;
41  import javax.lang.model.type.TypeMirror;
42  
43  import com.helger.jcodemodel.AbstractJClass;
44  import com.helger.jcodemodel.AbstractJType;
45  import com.helger.jcodemodel.JCodeModel;
46  import com.helger.jcodemodel.JDefinedClass;
47  import com.helger.jcodemodel.JMethod;
48  import com.helger.jcodemodel.JTypeVar;
49  import com.helger.jcodemodel.JVar;
50  
51  /**
52   * @author Victor Nazarov <asviraspossible@gmail.com>
53   */
54  class ClassFiller
55  {
56    private final JDefinedClass _newClass;
57    private final JCodeModel _codeModel;
58    private final DecidedErrorTypesModelsAdapter _modelsAdapter;
59  
60    ClassFiller (final JCodeModel codeModel,
61                 final DecidedErrorTypesModelsAdapter modelsAdapter,
62                 final JDefinedClass newClass)
63    {
64      this._codeModel = codeModel;
65      this._modelsAdapter = modelsAdapter;
66      this._newClass = newClass;
67    }
68  
69    void fillClass (final TypeElement element, final TypeEnvironment environment) throws CodeModelBuildingException
70    {
71      _newClass.hide ();
72      final Annotator classAnnotator = new Annotator (_modelsAdapter, _newClass, environment);
73      classAnnotator.annotate (element.getAnnotationMirrors ());
74      for (final TypeParameterElement parameter : element.getTypeParameters ())
75      {
76        final JTypeVar typeVariable = _newClass.generify (parameter.getSimpleName ().toString ());
77        environment.put (typeVariable.name (), typeVariable);
78        for (final TypeMirror type : parameter.getBounds ())
79        {
80          typeVariable.bound ((AbstractJClass) _modelsAdapter.toJType (type, environment));
81        }
82      }
83      final TypeMirror superclass = element.getSuperclass ();
84      if (superclass != null && superclass.getKind () != TypeKind.NONE)
85      {
86        _newClass._extends ((AbstractJClass) _modelsAdapter.toJType (superclass, environment));
87      }
88      for (final TypeMirror iface : element.getInterfaces ())
89      {
90        _newClass._implements ((AbstractJClass) _modelsAdapter.toJType (iface, environment));
91      }
92      for (final Element enclosedElement : element.getEnclosedElements ())
93      {
94        if (enclosedElement.getKind ().equals (ElementKind.INTERFACE) ||
95            enclosedElement.getKind ().equals (ElementKind.CLASS))
96        {
97          final TypeElement innerClassElement = (TypeElement) enclosedElement;
98          _modelsAdapter.defineInnerClass (_newClass, innerClassElement, environment.enclosed ());
99        }
100       else
101         if (enclosedElement.getKind ().equals (ElementKind.METHOD))
102         {
103           final ExecutableElement executable = (ExecutableElement) enclosedElement;
104           final JMethod method = _newClass.method (DecidedErrorTypesModelsAdapter.toJMod (executable.getModifiers ()),
105                                                    _codeModel.VOID,
106                                                    executable.getSimpleName ().toString ());
107           final TypeEnvironment methodEnvironment = environment.enclosed ();
108           final Annotator methodAnnotator = new Annotator (_modelsAdapter, method, environment);
109           methodAnnotator.annotate (executable.getAnnotationMirrors ());
110           for (final TypeParameterElement parameter : executable.getTypeParameters ())
111           {
112             final JTypeVar typeVariable = method.generify (parameter.getSimpleName ().toString ());
113             methodEnvironment.put (typeVariable.name (), typeVariable);
114             for (final TypeMirror type : parameter.getBounds ())
115             {
116               typeVariable.bound ((AbstractJClass) _modelsAdapter.toJType (type, methodEnvironment));
117             }
118           }
119           method.type (_modelsAdapter.toJType (executable.getReturnType (), methodEnvironment));
120           for (final TypeMirror type : executable.getThrownTypes ())
121           {
122             final AbstractJClass throwable = (AbstractJClass) _modelsAdapter.toJType (type, methodEnvironment);
123             method._throws (throwable);
124           }
125           final List <? extends VariableElement> parameters = executable.getParameters ();
126           int n = 0;
127           for (final VariableElement variable : parameters)
128           {
129             final String parameterName = variable.getSimpleName ().toString ();
130             final TypeMirror parameterTypeMirror = variable.asType ();
131             final AbstractJType parameterType = _modelsAdapter.toJType (parameterTypeMirror, methodEnvironment);
132             JVar param;
133             if (executable.isVarArgs () && n == parameters.size () - 1)
134             {
135               param = method.varParam (DecidedErrorTypesModelsAdapter.toJMod (variable.getModifiers ()),
136                                        parameterType.elementType (),
137                                        parameterName);
138             }
139             else
140             {
141               param = method.param (DecidedErrorTypesModelsAdapter.toJMod (variable.getModifiers ()),
142                                     parameterType,
143                                     parameterName);
144             }
145             final Annotator parametorAnnotator = new Annotator (_modelsAdapter, param, methodEnvironment);
146             parametorAnnotator.annotate (variable.getAnnotationMirrors ());
147             n++;
148           }
149         }
150     }
151   }
