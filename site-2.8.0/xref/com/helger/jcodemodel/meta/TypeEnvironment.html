30  package com.helger.jcodemodel.meta;
31  
32  import java.util.Map;
33  import java.util.TreeMap;
34  
35  import com.helger.jcodemodel.AbstractJType;
36  
37  /**
38   * @author Victor Nazarov <asviraspossible@gmail.com>
39   */
40  class TypeEnvironment
41  {
42    private final Map <String, AbstractJType> map = new TreeMap <String, AbstractJType> ();
43    private final TypeEnvironment _parent;
44  
45    TypeEnvironment ()
46    {
47      _parent = null;
48    }
49  
50    private TypeEnvironment (final TypeEnvironment parent)
51    {
52      this._parent = parent;
53    }
54  
55    public TypeEnvironment enclosed ()
56    {
57      return new TypeEnvironment (this);
58    }
59  
60    public AbstractJType get (final String name)
61    {
62      final AbstractJType result = map.get (name);
63      if (result != null || _parent == null)
64        return result;
65      return _parent.get (name);
66    }
67  
68    public void put (final String name, final AbstractJType type)
69    {
70      map.put (name, type);
71    }
72  }
