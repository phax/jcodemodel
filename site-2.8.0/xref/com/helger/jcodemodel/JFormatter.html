90      /**
91       * @return <code>true</code> if the short name is ambiguous in context of
92       *         enclosingClass and classes with this name can't be imported.
93       */
94      public boolean isAmbiguousIn (final JDefinedClass aEnclosingClass)
95      {
96        // more than one type with the same name
97        if (m_aReferencedClasses.size () > 1)
98          return true;
99  
100       // an id and (at least one) type with the same name
101       if (m_bIsVariableName && !m_aReferencedClasses.isEmpty ())
102         return true;
103 
104       // no references is always unambiguous
105       if (m_aReferencedClasses.isEmpty ())
106         return false;
107 
108       // we have exactly one reference
109       AbstractJClass aSingleRef = m_aReferencedClasses.get (0);
110       if (aSingleRef instanceof JAnonymousClass)
111       {
112         aSingleRef = ((JAnonymousClass) aSingleRef).base ();
113       }
114 
115       // special case where a generated type collides with a type in package
116       // java.lang
117       if (aSingleRef._package () == JFormatter.this.m_aPckJavaLang)
118       {
119         // make sure that there's no other class with this name within the
120         // same package
121         for (final JDefinedClass aClass : aEnclosingClass._package ().classes ())
122         {
123           // even if this is the only "String" class we use,
124           // if the class called "String" is in the same package,
125           // we still need to import it.
126           if (aClass.name ().equals (aSingleRef.name ()))
127           {
128             // collision -> ambiguous
129             return true;
130           }
131         }
132       }
133 
134       return false;
135     }

234     @Nonnull
235     private AbstractJClass _getClassForImport (@Nonnull final AbstractJClass aClass)
236     {
237       AbstractJClass aRealClass = aClass;
238       if (aRealClass instanceof JAnonymousClass)
239       {
240         // get the super class of the anonymous class
241         aRealClass = ((JAnonymousClass) aRealClass).base ();
242       }
243       if (aRealClass instanceof JNarrowedClass)
244       {
245         // Never imported narrowed class but the erasure only
246         aRealClass = aRealClass.erasure ();
247       }
248       return aRealClass;
249     }
250 
251     public boolean add (@Nonnull final AbstractJClass aClass)
252     {
253       final AbstractJClass aRealClass = _getClassForImport (aClass);
254 
255       // Avoid importing 2 classes with the same class name
256       if (!m_aNames.add (aRealClass.name ()))
257       {
258         if (m_bImportDebug)
259           System.out.println ("A class with local name '" + aRealClass.name () + "' is already in the import list.");
260         return false;
261       }
262 
263       if (!m_aClasses.add (aRealClass))
264       {
265         if (m_bImportDebug)
266           System.out.println ("The class '" + aRealClass.fullName () + "' is already in the import list.");
267         return false;
268       }
269 
270       if (m_bImportDebug)
271         System.out.println ("Added import class '" + aClass.fullName () + "'");
272       return true;
273     }
274 
275     public boolean contains (@Nullable final AbstractJClass aClass)
276     {
277       final AbstractJClass aRealClass = _getClassForImport (aClass);
278 
279       return m_aClasses.contains (aRealClass);
280     }

433   private static boolean _needSpace (final char c1, final char c2)
434   {
435     if ((c1 == ']') && (c2 == '{'))
436       return true;
437     if (c1 == ';')
438       return true;
439     if (c1 == CLOSE_TYPE_ARGS)
440     {
441       // e.g., "public Foo<Bar> test;"
442       if (c2 == '(')
443       {
444         // but not "new Foo<Bar>()"
445         return false;
446       }
447       return true;
448     }
449     if ((c1 == ')') && (c2 == '{'))
450       return true;
451     if ((c1 == ',') || (c1 == '='))
452       return true;
453     if (c2 == '=')
454       return true;
455     if (Character.isDigit (c1))
456     {
457       if ((c2 == '(') || (c2 == ')') || (c2 == ';') || (c2 == ','))
458         return false;
459       return true;
460     }
461     if (Character.isJavaIdentifierPart (c1))
462     {
463       switch (c2)
464       {
465         case '{':
466         case '}':
467         case '+':
468         case '-':
469         case '>':
470         case '@':
471           return true;
472         default:
473           return Character.isJavaIdentifierStart (c2);
474       }
475     }
476     if (Character.isJavaIdentifierStart (c2))
477     {
478       switch (c1)
479       {
480         case ']':
481         case ')':
482         case '}':
483         case '+':
484           return true;
485         default:
486           return false;
487       }
488     }
489     if (Character.isDigit (c2))
490     {
491       if (c1 == '(')
492         return false;
493       return true;
494     }
495     return false;
496   }

498   private void _spaceIfNeeded (final char c)
499   {
500     if (m_bAtBeginningOfLine)
501     {
502       for (int i = 0; i < m_nIndentLevel; i++)
503         m_aPW.print (m_sIndentSpace);
504       m_bAtBeginningOfLine = false;
505     }
506     else
507       if (m_cLastChar != 0 && _needSpace (m_cLastChar, c))
508         m_aPW.print (' ');
509   }

511   /**
512    * Print a char into the stream
513    *
514    * @param c
515    *        the char
516    * @return this for chaining
517    */
518   @Nonnull
519   public JFormatter print (final char c)
520   {
521     if (m_eMode == EMode.PRINTING)
522     {
523       if (c == CLOSE_TYPE_ARGS)
524       {
525         m_aPW.print ('>');
526       }
527       else
528       {
529         _spaceIfNeeded (c);
530         m_aPW.print (c);
531       }
532       m_cLastChar = c;
533     }
534     return this;
535   }

537   /**
538    * Print a String into the stream
539    *
540    * @param s
541    *        the String
542    * @return this
543    */
544   @Nonnull
545   public JFormatter print (@Nonnull final String s)
546   {
547     if (m_eMode == EMode.PRINTING && s.length () > 0)
548     {
549       _spaceIfNeeded (s.charAt (0));
550       m_aPW.print (s);
551       m_cLastChar = s.charAt (s.length () - 1);
552     }
553     return this;
554   }

564   /**
565    * Print a type name.
566    * <p>
567    * In the collecting mode we use this information to decide what types to
568    * import and what not to.
569    *
570    * @param aType
571    *        Type to be emitted
572    * @return this for chaining
573    */
574   @Nonnull
575   public JFormatter type (@Nonnull final AbstractJClass aType)
576   {
577     switch (m_eMode)
578     {
579       case COLLECTING:
580         final String sShortName = aType.name ();
581         NameUsage aUsages = m_aCollectedReferences.get (sShortName);
582         if (aUsages == null)
583         {
584           aUsages = new NameUsage (sShortName);
585           m_aCollectedReferences.put (sShortName, aUsages);
586         }
587         aUsages.addReferencedType (aType);
588         break;
589       case PRINTING:
590         // many of the JTypes in this list are either primitive or belong to
591         // package java so we don't need a FQCN
592         if (m_aImportedClasses.contains (aType) || aType._package () == m_aPckJavaLang)
593         {
594           // FQCN imported or not necessary, so generate short name
595           print (aType.name ());
596         }
597         else
598         {
599           final AbstractJClass aOuter = aType.outer ();
600           if (aOuter != null)
601           {
602             type (aOuter).print ('.').print (aType.name ());
603           }
604           else
605           {
606             // collision was detected, so generate FQCN
607             print (aType.fullName ());
608           }
609         }
610         break;
611       case FIND_ERROR_TYPES:
612         if (aType.isError ())
613           m_bContainsErrorTypes = true;
614         break;
615     }
616     return this;
617   }

619   /**
620    * Print an identifier
621    *
622    * @param id
623    *        identifier
624    * @return this for chaining
625    */
626   @Nonnull
627   public JFormatter id (@Nonnull final String id)
628   {
629     switch (m_eMode)
630     {
631       case COLLECTING:
632         // see if there is a type name that collides with this id
633         NameUsage aUsages = m_aCollectedReferences.get (id);
634         if (aUsages == null)
635         {
636           // not a type, but we need to create a place holder to
637           // see if there might be a collision with a type
638           aUsages = new NameUsage (id);
639           m_aCollectedReferences.put (id, aUsages);
640         }
641         aUsages.setVariableName ();
642         break;
643       case PRINTING:
644         print (id);
645         break;
646     }
647     return this;
648   }

750   private boolean _collectCausesNoAmbiguities (@Nonnull final AbstractJClass aReference,
751                                                @Nonnull final JDefinedClass aClassToBeWritten)
752   {
753     if (m_bImportDebug)
754       System.out.println ("_collectCausesNoAmbiguities(" +
755                           aReference.fullName () +
756                           ", " +
757                           aClassToBeWritten.fullName () +
758                           ")");
759 
760     final NameUsage aUsages = m_aCollectedReferences.get (aReference.name ());
761     if (aUsages == null)
762       return true;
763     return !aUsages.isAmbiguousIn (aClassToBeWritten) && aUsages.containsReferencedType (aReference);
764   }

766   /**
767    * determine if an import statement should be used for given class. This is a
768    * matter of style and convention
769    *
770    * @param aReference
771    *        {@link AbstractJClass} referenced class
772    * @param aClassToBeWritten
773    *        {@link AbstractJClass} currently generated class
774    * @return <code>true</code> if an import statement can be used to shorten
775    *         references to referenced class
776    */
777   private boolean _collectShouldBeImported (@Nonnull final AbstractJClass aReference,
778                                             @Nonnull final JDefinedClass aClassToBeWritten)
779   {
780     if (m_bImportDebug)
781       System.out.println ("_collectShouldBeImported(" +
782                           aReference.fullName () +
783                           ", " +
784                           aClassToBeWritten.fullName () +
785                           ")");
786 
787     AbstractJClass aRealReference = aReference;
788     if (aRealReference instanceof JAnonymousClass)
789     {
790       // get the super class of the anonymous class
791       aRealReference = ((JAnonymousClass) aRealReference).base ();
792     }
793     if (aRealReference instanceof JNarrowedClass)
794     {
795       // Remove the generic arguments
796       aRealReference = aRealReference.erasure ();
797     }
798 
799     // Is it an inner class?
800     final AbstractJClass aOuter = aRealReference.outer ();
801     if (aOuter != null)
802     {
803       // Import inner class only when it's name contain a name of enclosing
804       // class.
805       // In such case no information is lost when we refer to inner class
806       // without mentioning it's enclosing class
807       if (aRealReference.name ().contains (aOuter.name ()))
808       {
809         // Recurse
810         if (_collectShouldBeImported (aOuter, aClassToBeWritten))
811           return true;
812       }
813 
814       // Do not import inner classes in all other cases to aid
815       // understandability/readability.
816       return false;
817     }
818     return true;
819   }
820 
821   /**
822    * If reference is inner-class adds some outer class to the list of imported
823    * classes if it
824    *
825    * @param aClassToBeWritten
826    *        {@link AbstractJClass} that may or may not have an import
827    * @param aGeneratingClass
828    *        {@link AbstractJClass} that is the current class being processed
829    * @return true if an import statement should be suppressed, false otherwise
830    */
831   private void _collectImportOuterClassIfCausesNoAmbiguities (@Nonnull final AbstractJClass aReference,
832                                                               @Nonnull final JDefinedClass aClassToBeWritten)
833   {
834     if (m_bImportDebug)
835       System.out.println ("_collectImportOuterClassIfCausesNoAmbiguities(" +
836                           aReference.fullName () +
837                           ", " +
838                           aClassToBeWritten.fullName () +
839                           ")");
840 
841     final AbstractJClass aOuter = aReference.outer ();
842     if (aOuter != null)
843     {
844       if (_collectCausesNoAmbiguities (aOuter, aClassToBeWritten) &&
845           _collectShouldBeImported (aOuter, aClassToBeWritten))
846       {
847         m_aImportedClasses.add (aOuter);
848       }
849       else
850       {
851         // Recursive call
852         _collectImportOuterClassIfCausesNoAmbiguities (aOuter, aClassToBeWritten);
853       }
854     }
855   }
856 
857   /**
858    * determine if an import statement should be suppressed
859    *
860    * @param aReference
861    *        {@link AbstractJClass} that may or may not have an import
862    * @param aGeneratingClass
863    *        {@link AbstractJClass} that is the current class being processed
864    * @return true if an import statement should be suppressed, false otherwise
865    */
866   private boolean _printIsImplicitlyImported (@Nonnull final AbstractJClass aReference,
867                                               @Nonnull final AbstractJClass aClassToBeWrittem)
868   {
869     if (m_bImportDebug)
870       System.out.println ("_printIsImplicitlyImported(" +
871                           aReference.fullName () +
872                           ", " +
873                           aClassToBeWrittem.fullName () +
874                           ")");
875 
876     AbstractJClass aRealReference = aReference;
877     if (aRealReference instanceof JAnonymousClass)
878     {
879       // Get the super class of the anonymous class
880       aRealReference = ((JAnonymousClass) aRealReference).base ();
881     }
882     if (aRealReference instanceof JNarrowedClass)
883     {
884       // Remove generic type arguments
885       aRealReference = aRealReference.erasure ();
886     }
887 
888     final JPackage aPackage = aRealReference._package ();
889     if (aPackage == null)
890     {
891       // May be null for JTypeVar and JTypeWildcard
892       return true;
893     }
894 
895     if (aPackage.isUnnamed ())
896     {
897       // Root package - no need to import something
898       return true;
899     }
900 
901     if (aPackage == m_aPckJavaLang)
902     {
903       // no need to explicitly import java.lang classes
904       return true;
905     }
906 
907     // All pkg local classes do not need an
908     // import stmt for ref, except for inner classes
909     if (aPackage == aClassToBeWrittem._package ())
910     {
911       AbstractJClass aOuter = aRealReference.outer ();
912       if (aOuter == null) // top-level class
913       {
914         // top-level package-local class needs no explicit import
915         return true;
916       }
917 
918       // inner-class
919       AbstractJClass aTopLevelClass = aOuter;
920       aOuter = aTopLevelClass.outer ();
921       while (aOuter != null)
922       {
923         aTopLevelClass = aOuter;
924         aOuter = aTopLevelClass.outer ();
925       }
926 
927       // if reference is inner-class and
928       // reference's top-level class is generated clazz,
929       // i. e. reference is enclosed in generated clazz,
930       // then it needs no explicit import statement.
931       return aTopLevelClass == aClassToBeWrittem;
932     }
933     return false;
934   }

936   /**
937    * Generates the whole source code out of the specified class.
938    *
939    * @param aClassToBeWritten
940    *        Class to be written
941    */
942   void write (@Nonnull final JDefinedClass aClassToBeWritten)
943   {
944     m_aPckJavaLang = aClassToBeWritten.owner ()._package ("java.lang");
945 
946     // first collect all the types and identifiers
947     m_eMode = EMode.COLLECTING;
948     m_aCollectedReferences.clear ();
949     m_aImportedClasses.clear ();
950     declaration (aClassToBeWritten);
951 
952     if (m_bImportDebug)
953       System.out.println ("***Start collecting***");
954 
955     // the class itself that we will be generating is always accessible and must
956     // be the first import
957     m_aImportedClasses.add (aClassToBeWritten);
958 
959     // collate type names and identifiers to determine which types can be
960     // imported
961     for (final NameUsage aUsage : m_aCollectedReferences.values ())
962     {
963       if (!aUsage.isAmbiguousIn (aClassToBeWritten) && !aUsage.isVariableName ())
964       {
965         final AbstractJClass aReferencedClass = aUsage.getSingleReferencedType ();
966 
967         if (_collectShouldBeImported (aReferencedClass, aClassToBeWritten))
968         {
969           m_aImportedClasses.add (aReferencedClass);
970         }
971         else
972         {
973           _collectImportOuterClassIfCausesNoAmbiguities (aReferencedClass, aClassToBeWritten);
974         }
975       }
976       else
977       {
978         if (aUsage.isTypeName ())
979           for (final AbstractJClass reference : aUsage.getReferencedTypes ())
980           {
981             _collectImportOuterClassIfCausesNoAmbiguities (reference, aClassToBeWritten);
982           }
983       }
984     }
985 
986     if (m_bImportDebug)
987       System.out.println ("***Finished collecting***");
988 
989     // then print the declaration
990     m_eMode = EMode.PRINTING;
991 
992     assert aClassToBeWritten.parentContainer ().isPackage () : "this method is only for a pacakge-level class";
993     final JPackage aPackage = (JPackage) aClassToBeWritten.parentContainer ();
994     if (!aPackage.isUnnamed ())
995     {
996       newline ().declaration (aPackage);
997       newline ();
998     }
999 
1000     // generate import statements
1001     boolean bAnyImport = false;
1002     for (final AbstractJClass aImportClass : m_aImportedClasses.getAllSorted ())
1003     {
1004       // suppress import statements for primitive types, built-in types,
1005       // types in the root package, and types in
1006       // the same package as the current type
1007       if (!_printIsImplicitlyImported (aImportClass, aClassToBeWritten))
1008       {
1009         print ("import").print (aImportClass.fullName ()).print (';').newline ();
1010         bAnyImport = true;
1011 
1012         if (m_bImportDebug)
1013           System.out.println ("  import " + aImportClass.fullName ());
1014       }
1015     }
1016 
1017     if (bAnyImport)
1018       newline ();
1019 
1020     declaration (aClassToBeWritten);
1021   }

1023   public static boolean containsErrorTypes (@Nonnull final JDefinedClass c)
1024   {
1025     final JFormatter aFormatter = new JFormatter (NullWriter.getInstance ());
1026     aFormatter.m_eMode = EMode.FIND_ERROR_TYPES;
