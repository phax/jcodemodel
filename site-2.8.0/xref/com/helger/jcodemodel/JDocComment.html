51  /**
52   * JavaDoc comment.
53   * <p>
54   * A javadoc comment consists of multiple parts. There's the main part (that
55   * comes the first in in the comment section), then the parameter parts
56   * (@param), the return part (@return), and the throws parts (@throws). TODO: it
57   * would be nice if we have JComment class and we can derive this class from
58   * there.
59   */
60  public class JDocComment extends JCommentPart implements IJGenerable, IJOwned
61  {
62    private static final String INDENT = " *     ";
63  
64    private static final long serialVersionUID = 1L;
65  
66    private final JCodeModel m_aOwner;
67  
68    /** list of @param tags */
69    private final Map <String, JCommentPart> m_aAtParams = new LinkedHashMap <String, JCommentPart> ();
70  
71    /** list of xdoclets */
72    private final Map <String, Map <String, String>> m_aAtXdoclets = new LinkedHashMap <String, Map <String, String>> ();
73  
74    /** list of @throws tags */
75    private final Map <AbstractJClass, JCommentPart> m_aAtThrows = new LinkedHashMap <AbstractJClass, JCommentPart> ();
76  
77    /**
78     * The @return tag part.
79     */
80    private JCommentPart m_aAtReturn;
81  
82    /**
83     * The @author tag part.
84     */
85    private JCommentPart m_aAtAuthor;
86  
87    /** The @deprecated tag */
88    private JCommentPart m_aAtDeprecated;
89  
90    protected JDocComment (@Nonnull final JCodeModel owner)
91    {
92      m_aOwner = JCValueEnforcer.notNull (owner, "Owner");
93    }
94  
95    @Nonnull
96    public JCodeModel owner ()
97    {
98      return m_aOwner;
99    }

101   @Override
102   public JDocComment append (@Nullable final Object o)
103   {
104     add (o);
105     return this;
106   }

108   /**
109    * Append a text to a @param tag to the javadoc
110    *
111    * @param param
112    *        Parameter to be added
113    * @return The created {@link JCommentPart}
114    */
115   @Nonnull
116   public JCommentPart addParam (final String param)
117   {
118     JCommentPart p = m_aAtParams.get (param);
119     if (p == null)
120     {
121       p = new JCommentPart ();
122       m_aAtParams.put (param, p);
123     }
124     return p;
125   }

132    * @return The created {@link JCommentPart}
133    */
134   public JCommentPart addParam (@Nonnull final JVar param)
135   {
136     return addParam (param.name ());
137   }

281   /**
282    * add an xdoclet.
283    *
284    * @param name
285    *        xdoclet name
286    * @return Map with the key/value pairs
287    */
288   @Nonnull
289   public Map <String, String> addXdoclet (final String name)
290   {
291     Map <String, String> p = m_aAtXdoclets.get (name);
292     if (p == null)
293     {
294       p = new LinkedHashMap <String, String> ();
295       m_aAtXdoclets.put (name, p);
296     }
297     return p;
298   }

300   /**
301    * add an xdoclet.
302    *
303    * @param name
304    *        xdoclet name
305    * @param attributes
306    *        Attributes to be added
307    * @return Map with the key/value pairs
308    */
309   @Nonnull
310   public Map <String, String> addXdoclet (final String name, final Map <String, String> attributes)
311   {
312     final Map <String, String> p = addXdoclet (name);
313     p.putAll (attributes);
314     return p;
315   }

317   /**
318    * add an xdoclet.
319    *
320    * @param name
321    *        xdoclet name
322    * @param attribute
323    *        Attribute name to be added
324    * @param value
325    *        Attribute value to be added
326    * @return Map with the key/value pairs
327    */
328   @Nonnull
329   public Map <String, String> addXdoclet (final String name, final String attribute, final String value)
330   {
331     final Map <String, String> p = addXdoclet (name);
332     p.put (attribute, value);
333     return p;
334   }

347   public void generate (@Nonnull final JFormatter f)
348   {
349     // I realized that we can't use StringTokenizer because
350     // this will recognize multiple \n as one token.
351 
352     f.print ("/**").newline ();
353 
354     format (f, " * ");
355 
356     f.print (" * ").newline ();
357     for (final Map.Entry <String, JCommentPart> e : m_aAtParams.entrySet ())
358     {
359       f.print (" * @param ").print (e.getKey ()).newline ();
360       e.getValue ().format (f, INDENT);
361     }
362     if (m_aAtReturn != null)
363     {
364       f.print (" * @return").newline ();
365       m_aAtReturn.format (f, INDENT);
366     }
367     if (m_aAtAuthor != null)
368     {
369       f.print (" * @author").newline ();
370       m_aAtAuthor.format (f, INDENT);
371     }
372     for (final Map.Entry <AbstractJClass, JCommentPart> e : m_aAtThrows.entrySet ())
373     {
374       f.print (" * @throws ").type (e.getKey ()).newline ();
375       e.getValue ().format (f, INDENT);
376     }
377     if (m_aAtDeprecated != null)
378     {
379       f.print (" * @deprecated").newline ();
380       m_aAtDeprecated.format (f, INDENT);
381     }
382     for (final Map.Entry <String, Map <String, String>> e : m_aAtXdoclets.entrySet ())
383     {
384       f.print (" * @").print (e.getKey ());
385       if (e.getValue () != null)
386       {
387         for (final Map.Entry <String, String> a : e.getValue ().entrySet ())
388         {
389           f.print (" ").print (a.getKey ()).print ("= \"").print (a.getValue ()).print ("\"");
390         }
391       }
392       f.newline ();
393     }
394     f.print (" */").newline ();
395   }
