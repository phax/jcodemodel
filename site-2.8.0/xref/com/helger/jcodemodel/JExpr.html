388   /**
389    * Escapes the given string, then surrounds it by the specified quotation
390    * mark.
391    *
392    * @param quote
393    *        Quote char. Either single quote (') or double quote (")
394    * @param s
395    *        Source string to quote
396    * @return Qutoed string
397    */
398   @Nonnull
399   public static String quotify (final char quote, @Nonnull final String s)
400   {
401     final int n = s.length ();
402     final StringBuilder sb = new StringBuilder (n + 2);
403     sb.append (quote);
404     for (int i = 0; i < n; i++)
405     {
406       final char c = s.charAt (i);
407       final int j = CHAR_ESCAPE.indexOf (c);
408       if (j >= 0)
409       {
410         if ((quote == '"' && c == '\'') || (quote == '\'' && c == '"'))
411         {
412           sb.append (c);
413         }
414         else
415         {
416           sb.append ('\\');
417           sb.append (CHAR_MACRO.charAt (j));
418         }
419       }
420       else
421       {
422         // technically Unicode escape shouldn't be done here,
423         // for it's a lexical level handling.
424         //
425         // However, various tools are so broken around this area,
426         // so just to be on the safe side, it's better to do
427         // the escaping here (regardless of the actual file encoding)
428         //
429         // see bug
430         if (c < 0x20 || c > 0x7E)
431         {
432           // not printable. use Unicode escape
433           sb.append ("\\u");
434           final String hex = Integer.toHexString (c & 0xFFFF);
435           for (int k = hex.length (); k < 4; k++)
436             sb.append ('0');
437           sb.append (hex);
438         }
439         else
440         {
441           sb.append (c);
442         }
443       }
444     }
445     sb.append (quote);
446     return sb.toString ();
447   }

449   @Nonnull
450   public static JAtom lit (final char c)

461   /**
462    * Creates an expression directly from a source code fragment.
463    * <p>
464    * This method can be used as a short-cut to create a JExpression. For
465    * example, instead of <code>_a.gt(_b)</code>, you can write it as:
466    * <code>JExpr.direct("a&gt;b")</code>.
467    * <p>
468    * Be warned that there is a danger in using this method, as it obfuscates the
469    * object model.
470    * 
471    * @param source
472    *        Java source code
473    * @return Direct expression
474    */
475   @Nonnull
476   public static AbstractJExpressionImpl direct (@Nonnull final String source)
477   {
478     return new AbstractJExpressionImpl ()
479     {
480       public void generate (final JFormatter f)
481       {
482         f.print ('(').print (source).print (')');
483       }
484     };
485   }
